Change 3: User Initials Background - Match Color Scheme
Update user avatar/initials component:
javascript// Your 5 brand colors (adjust to your actual colors)
const brandColors = [
  '#8B4789', // Plum
  '#D4A5D4', // Light purple
  '#6B4C6B', // Dark plum
  '#A87CA8', // Medium purple
  '#E5C5E5', // Very light purple
];

// Function to get consistent color for a user
const getUserColor = (userId) => {
  // Use user ID to consistently assign one of the 5 colors
  const index = userId.charCodeAt(0) % brandColors.length;
  return brandColors[index];
};

// In your user avatar component
const UserAvatar = ({ user }) => {
  const initials = user.name
    .split(' ')
    .map(n => n[0])
    .join('')
    .toUpperCase();
  
  const backgroundColor = getUserColor(user.id);
  
  return (
    <div 
      className="user-avatar"
      style={{
        backgroundColor: backgroundColor,
        color: '#FFFFFF',
        width: '40px',
        height: '40px',
        borderRadius: '50%',
        display: 'flex',
        alignItems: 'center',
        justifyContent: 'center',
        fontWeight: 'bold',
        fontSize: '16px'
      }}
    >
      {initials}
    </div>
  );
};
