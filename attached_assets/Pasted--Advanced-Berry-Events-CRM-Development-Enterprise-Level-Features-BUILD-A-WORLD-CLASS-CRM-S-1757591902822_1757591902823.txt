### Database Schema Enhancement
```sql
-- Enhanced CRM Tables Structure
CREATE TABLE contacts (
  id UUID PRIMARY KEY,
  type ENUM('consumer', 'provider'),
  first_name VARCHAR(100),
  last_name VARCHAR(100),
  email VARCHAR(255) UNIQUE,
  phone VARCHAR(20),
  lead_score INTEGER DEFAULT 0,
  lifecycle_stage VARCHAR(50),
  created_at TIMESTAMP,
  updated_at TIMESTAMP
);

CREATE TABLE opportunities (
  id UUID PRIMARY KEY,
  contact_id UUID REFERENCES contacts(id),
  stage VARCHAR(50),
  value DECIMAL(10,2),
  probability INTEGER,
  expected_close_date DATE,
  source VARCHAR(100),
  created_at TIMESTAMP
);

CREATE TABLE activities (
  id UUID PRIMARY KEY,
  contact_id UUID REFERENCES contacts(id),
  type VARCHAR(50),
  subject VARCHAR(255),
  description TEXT,
  completed BOOLEAN DEFAULT FALSE,
  due_date TIMESTAMP,
  created_by UUID,
  created_at TIMESTAMP
);

CREATE TABLE campaigns (
  id UUID PRIMARY KEY,
  name VARCHAR(255),
  type VARCHAR(50),
  status VARCHAR(50),
  start_date DATE,
  end_date DATE,
  budget DECIMAL(10,2),
  created_at TIMESTAMP
);
```
