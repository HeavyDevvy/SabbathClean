javascript// File: src/pages/BookingSteps.jsx or src/components/BookingFlow.jsx

import { useLocation } from 'react-router-dom';
import { useState, useEffect } from 'react';

const BookingSteps = () => {
  const location = useLocation();
  const [currentStep, setCurrentStep] = useState(1);
  const [bookingData, setBookingData] = useState({});
  
  // Get pre-selected provider from navigation state
  const preSelectedProvider = location.state?.preSelectedProvider;

  useEffect(() => {
    if (preSelectedProvider) {
      console.log('Provider pre-selected from Berry Stars:', preSelectedProvider.name);
      setBookingData(prev => ({
        ...prev,
        selectedProvider: preSelectedProvider,
        providerLocked: true,
      }));
    }
  }, [preSelectedProvider]);

  // Step 4: Provider Selection
  const ProviderSelectionStep = () => {
    // If provider is locked (from Berry Stars)
    if (bookingData.providerLocked && bookingData.selectedProvider) {
      return (
        <div className="provider-selection-locked" style={{
          textAlign: 'center',
          padding: '2rem',
        }}>
          <h2 style={{ color: '#2E0B2B', marginBottom: '2rem' }}>

            </div>
            
            <div className="specialties" style={{

          </div>

            <button 
              onClick={() => setCurrentStep(currentStep - 1)}

                padding: '0.75rem 1.5rem',

                border: '2px solid #8B4789',
                color: '#8B4789',
                borderRadius: '8px',

              backgroundColor: '#FFFFFF',
              border: '2px solid #8B4789',

              border: 'none',
              borderRadius: '8px',
              cursor: bookingData.selectedProvider ? 'pointer' : 'not-allowed',

          >

  // Rest of booking flow...
  const renderCurrentStep = () => {
    switch(currentStep) {
      case 1: return <ServiceSelectionStep />;
      case 2: return <ConfigurationStep />;
      case 3: return <DateTimeStep />;
      case 4: return <ProviderSelectionStep />;

    <div className="booking-flow">

        width: '100%',
        height: '6px',

          width: `${(currentStep / 8) * 100}%`,
          height: '100%',
          backgroundColor: '#8B4789',
          transition: 'width 0.4s ease',
        }} />
      </div>
      
      {renderCurrentStep()}
    </div>
  );
};
